// Firestore Security Rules for XMAS Pumpfun Race
// Go to Firebase Console > Firestore Database > Rules and paste:

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Chat messages - anyone can read, authenticated users can write
    match /chat_messages/{messageId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Live bets - anyone can read, authenticated users can write
    match /live_bets/{betId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Profiles - anyone can read, users can write their own
    match /profiles/{walletId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Wallets collection
    match /wallets/{walletId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Jackpot state - tracks current race and next jackpot
    match /jackpot_state/{stateId} {
      allow read: if true;
      allow write: if true;
    }
    
    // Jackpot winners history
    match /jackpot_winners/{winnerId} {
      allow read: if true;
      allow write: if true;
    }
  }
}
